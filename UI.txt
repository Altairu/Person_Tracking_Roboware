            <button class="button" data-direction="FORWARD">↑ Forward</button>
            <button class="button" data-direction="STOP">⏹ Stop</button>
            <button class="button" data-direction="BACKWARD">↓ Backward</button>

    <script>
        const ws = new WebSocket("ws://192.168.2.212:8080/ws");
        const modeButton = document.getElementById("mode-button");

        let currentMode = "CONTROL"; // 初期モード

        modeButton.addEventListener("click", () => {
            currentMode = currentMode === "CONTROL" ? "FOLLOW" : "CONTROL";
            modeButton.textContent = currentMode === "CONTROL" ? "Switch to Follow Mode" : "Switch to Control Mode";
            ws.send(currentMode);
        });

        document.querySelectorAll(".direction-buttons .button").forEach(btn => {
            btn.addEventListener("click", () => {
                const direction = btn.getAttribute("data-direction");
                ws.send(direction);
            });
        });

        ws.onmessage = event => {
            console.log("Received from server:", event.data);
        };
    </script>
